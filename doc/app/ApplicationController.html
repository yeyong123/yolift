def set_local
        if params[:locale]
                if I18n.available_locales.include?(params[:locale].to_sym)
                        I18n.locale = params[:locale]
                else
                        flash.now[:notice] = "#{params[:locale]} 翻译不可用。"
                        logger.error flash.now[:notice]
                end
        end
end
